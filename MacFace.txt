program MacFace;{ MacFace v1.00 }{ Copyleft 2022 by Erich Kohl }{ Just a tiny graphics demo that }{ draws a face that looks similar }{ to the Finder face. }{ Hello } var  box : rect;  a : integer;  b : boolean; procedure MouthClosed; begin  moveto(150, 255);  lineto(250, 255); end; procedure MouthOpen;  var   a : integer; begin  for a := 1 to 10 do   begin    box.top := 255 - a;    box.bottom := 255 + a;    box.left := 200 - a;    box.right := 200 + a;    frameoval(box);   end; end; function Pause : boolean;  var   a : integer;   event : eventrecord;   b : boolean;   quit : boolean; begin  quit := false;  a := 1;  repeat   b := GetNextEvent(keyDownMask, event);   if (b = true) and (event.what = 3) and (event.message mod 256 = 27) then    begin     quit := true;    end;   a := a + 1;  until (a = 200) or (quit = true);  Pause := quit; end;begin{ Draw the eyes } box.top := 30; box.bottom := 80; box.left := 100; box.right := 150; frameoval(box); box.left := 250; box.right := 300; frameoval(box); for a := 1 to 5 do  begin   box.top := 55 - a;   box.bottom := 55 + a;   box.left := 125 - a;   box.right := 125 + a;   frameoval(box);   box.left := 275 - a;   box.right := 275 + a;   frameoval(box);  end;{ Draw the nose } moveto(200, 55); lineto(175, 180); lineto(200, 180); lineto(200, 195);{ Now repeatedly open and close }{ the mouth until user hits Esc } repeat  penpat(black);  MouthClosed;  b := Pause;  if (b = false) then   begin    penpat(white);    MouthClosed;    penpat(black);    MouthOpen;    b := Pause;    penpat(white);    MouthOpen;   end; until (b = true);end.
